Printing Multiple Ints in a String

#include <stdio.h>

int main(void) {
    int num1, num2, num3;

    num1 = 1;
    num2 = 2;
    num3 = 3;

    printf("%d plus %d is equal to %d\n", num1, num2, num3);

    return 0;
}
